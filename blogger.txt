import cv2
import mediapipe as mp
# استدعاء صريح ومباشر للموديولات الفرعية لتجنب خطأ AttributeError تماماً
import mediapipe.python.solutions.hands as mp_hands
import mediapipe.python.solutions.drawing_utils as mp_draw

# إعدادات نموذج تتبع اليد
hands = mp_hands.Hands(
    static_image_mode=False,
    max_num_hands=2,
    min_detection_confidence=0.7,
    min_tracking_confidence=0.5
)

# تشغيل الكاميرا الافتراضية 
# (إذا كنت تستخدم نظام وهمي، تأكد من عمل Pass-through لكاميرا الـ USB الخارجي)
cap = cv2.VideoCapture(0)

print("[+] تم تشغيل الكاميرا بنجاح داخل بيئة كالي. اضغط على 'q' للخروج.")

while cap.isOpened():
    success, frame = cap.read()
    if not success:
        print("[!] فشل في التقاط الإطار من الكاميرا. تحقق من اتصالها.")
        break

    # قلب الصورة أفقياً لمحاكاة حركة المرآة الطبيعية
    frame = cv2.flip(frame, 1)

    # تحويل الألوان من BGR إلى RGB لكي يفهمها نموذج ميديا بايب
    img_rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)

    # معالجة الإطار واستخراج نقاط اليد
    results = hands.process(img_rgb)

    # التحقق من وجود يد في الإطار
    if results.multi_hand_landmarks:
        for hand_landmarks in results.multi_hand_landmarks:
            
            # استخراج أبعاد الشاشة الحالية بالبكسل لتحديد المواقع بدقة
            h, w, c = frame.shape
            
            # حلقة لتتبع الـ 21 نقطة ومفصل في اليد
            for id, lm in enumerate(hand_landmarks.landmark):
                # تحويل الإحداثيات النسبية الحساسة إلى نقاط بكسل فعلية على الشاشة
                cx, cy = int(lm.x * w), int(lm.y * h)
                
                # طباعة إحداثيات مفصل المعصم (النقطة رقم 0) كمثال في الترمينال
                if id == 0:
                    print(f"[TRACKING] Wrist -> X: {cx}, Y: {cy}")

            # رسم الهيكل العظمي والنقاط لليد فوق البث المباشر
            mp_draw.draw_landmarks(
                frame, 
                hand_landmarks, 
                mp_hands.HAND_CONNECTIONS,
                mp_draw.DrawingSpec(color=(0, 0, 255), thickness=2, circle_radius=4), # نقاط حمراء للمفاصل
                mp_draw.DrawingSpec(color=(255, 0, 0), thickness=2)  # خطوط زرقاء بين المفاصل
            )

    # عرض النافذة للمستخدم
    cv2.imshow("CyberSecurity Lab - Hand Tracking", frame)

    # الخروج الفوري عند الضغط على مفتاح 'q' من الكيبورد
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

# تنظيف الذاكرة وإغلاق الكاميرا والنوافذ بعد الإنهاء
cap.release()
cv2.destroyAllWindows()
