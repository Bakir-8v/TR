#!/bin/bash

# ألوان للتنظيم
GREEN='\033[0;32m'
RED='\033[0;31m'
BLUE='\033[0;34m'
YELLOW='\033[0;33m'
NC='\033[0m'

TARGET="https://www.syria.tv"

echo -e "${BLUE}[*] بدء محاولة استخراج إصدارات الموديلات عبر ملفات التعريف...${NC}"

# مصفوفة تحتوي على اسم الموديل والمسار المتوقع لملف التعريف الخاص به
declare -A MOD_FILES
MOD_FILES=(
    ["blazy"]="modules/contrib/blazy/blazy.info.yml"
    ["better_exposed_filters"]="modules/contrib/better_exposed_filters/better_exposed_filters.info.yml"
    ["extlink"]="modules/contrib/extlink/extlink.info.yml"
    ["lazy"]="modules/contrib/lazy/lazy.info.yml"
)

for MOD in "${!MOD_FILES[@]}"
do
    URL="${TARGET}/${MOD_FILES[$MOD]}"
    STATUS=$(curl -s -o /dev/null -w "%{http_code}" "$URL")
    
    if [ "$STATUS" -eq 200 ]; then
        echo -e "${GREEN}[+] تم الوصول لملف التعريف لـ ($MOD) - الحالة $STATUS${NC}"
        echo -e "${BLUE}--- تفاصيل الإصدار المستخرج ---${NC}"
        # محاولة قراءة الأسطر التي تحتوي على الإصدار أو التسمية
        curl -s "$URL" | grep -E "version:|project:" || echo "تعذر تصفية النص تلقائياً."
        echo -e "${BLUE}--------------------------------${NC}\n"
    else
        echo -e "${RED}[-] ملف التعريف محجوب أو غير موجود لـ ($MOD) - الحالة $STATUS${NC}"
    fi
done

echo -e "\n${YELLOW}[*] تم الانتهاء من محاولة جلب الإصدارات.${NC}"
