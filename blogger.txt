#!/bin/bash

# ألوان لتنظيم المخرجات
GREEN='\033[0;32m'
RED='\033[0;31m'
BLUE='\033[0;34m'
NC='\033[0m' # No Color

echo -e "${BLUE}[*] بدء سكربت فحص ومطابقة ثغرات إصدارات دروبال...${NC}"

# 1. التحقق من وجود أداة searchsploit
if ! command -v searchsploit &> /dev/null
then
    echo -e "${RED}[!] أداة searchsploit غير مثبتة. يرجى تثبيتها أو تحديث قاعدة بيانات ExploitDB.${NC}"
    exit 1
fi

# 2. البحث عن ثغرات النواة المكتشفة (Drupal 9.5.7)
echo -e "\n${GREEN}[+] فحص ثغرات النواة لـ Drupal 9.x...${NC}"
searchsploit drupal 9.

# 3. البحث عن ثغرات مكتبة jQuery UI 1.13.2
echo -e "\n${GREEN}[+] فحص ثغرات المكتبة المرفقة jQuery UI 1.13.2...${NC}"
searchsploit jquery ui 1.13

# 4. محاولة فحص المسارات الافتراضية الحساسة لدروبال (Reconnaissance)
# يمكنك استبدال target.com برابط الهدف الفعلي عند التشغيل
echo -e "\n${GREEN}[+] قائمة بالمسارات الافتراضية التي يفضل التحقق منها يدوياً أو عبر curl:${NC}"
echo -e "${BLUE}- /core/changelog.txt${NC} (لتأكيد الإصدار الدقيق لنواة النظام)"
echo -e "${BLUE}- /composer.json${NC} (لقراءة المكتبات البرمجية المعتمدة في الخلفية)"
echo -e "${BLUE}- /web.config أو /robots.txt${NC} (لمعرفة المجلدات المستثناة من الأرشفة)"

echo -e "\n${BLUE}[*] انتهى الفحص الأولي. يمكنك الآن مطابقة الـ CVEs المستخرجة مع بيئة التجارب الخاصة بك.${NC}"
