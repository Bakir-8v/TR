import cv2
import mediapipe as mp
from mediapipe.tasks import python
from mediapipe.tasks.python import vision
import numpy as np

# 1. إعداد خيارات ميديا بايب الحديثة
BaseOptions = mp.tasks.BaseOptions
HandLandmarker = mp.tasks.vision.HandLandmarker
HandLandmarkerOptions = mp.tasks.vision.HandLandmarkerOptions
VisionRunningMode = mp.tasks.vision.RunningMode

# تشغيل الكاميرا
cap = cv2.VideoCapture(0)

# 2. بناء دالة برمجية لاستقبال ومعالجة إطارات الفيديو فوراً
def print_result(result: vision.HandLandmarkerResult, output_image: mp.Image, timestamp_ms: int):
    if result.hand_landmarks:
        for hand_landmarks in result.hand_landmarks:
            # طباعة إحداثيات المفصل الأول (المعصم) كمثال
            wrist = hand_landmarks[0]
            print(f"[TRACKING] Wrist -> X: {wrist.x:.2f}, Y: {wrist.y:.2f}")

# إعدادات المكتشف (تنبيه: سيعتمد على المعالجة المباشرة LIVE_STREAM)
options = HandLandmarkerOptions(
    base_options=BaseOptions(model_asset_buffer=None), # سنعتمد على معالجة الإطارات العادية
    running_mode=VisionRunningMode.VIDEO,
    num_hands=2
)

print("[+] تم تشغيل النموذج الحديث بنجاح. اضغط على 'q' للخروج.")

# لتسهيل الأمر وتجنب تحميل ملفات .task ضخمة، سنستخدم الحل الهجين المستقر:
mp_hands = mp.solutions.hands if hasattr(mp, 'solutions') else None

if mp_hands is None:
    # إذا كانت الحزمة مكسورة تماماً في النظام، هذا الكود الالتفافي (Fallback) سيعيد بناء المسار برمجياً
    import sys
    for path in sys.path:
        if 'site-packages' in path:
            sys.path.append(f"{path}/mediapipe/modules")

# العودة للطريقة الكلاسيكية الآمنة ولكن مع معالجة الاستدعاء عبر السلسلة النصية المباشرة (Dynamic Import)
import importlib
try:
    sub_hands = importlib.import_module('mediapipe.python.solutions.hands')
    sub_draw = importlib.import_module('mediapipe.python.solutions.drawing_utils')
except ModuleNotFoundError:
    # الطريقة الأخيرة الممكنة للاستدعاء في بعض توزيعات دبيان المخصصة
    import mediapipe.framework.formats.landmark_pb2 as landmark_pb2
    sub_hands = mp.solutions.hands
    sub_draw = mp.solutions.drawing_utils

hands = sub_hands.Hands(static_image_mode=False, max_num_hands=2, min_detection_confidence=0.7)

while cap.isOpened():
    success, frame = cap.read()
    if not success: break
    
    frame = cv2.flip(frame, 1)
    img_rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
    results = hands.process(img_rgb)
    
    if results.multi_hand_landmarks:
        for hand_landmarks in results.multi_hand_landmarks:
            sub_draw.draw_landmarks(frame, hand_landmarks, sub_hands.HAND_CONNECTIONS)
            
    cv2.imshow("Hand Tracking - Stable Mode", frame)
    if cv2.waitKey(1) & 0xFF == ord('q'): break

cap.release()
cv2.destroyAllWindows()
