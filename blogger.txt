#!/bin/bash

# ألوان للتنظيم
GREEN='\033[0;32m'
RED='\033[0;31m'
BLUE='\033[0;34m'
YELLOW='\033[0;33m'
NC='\033[0m'

echo -e "${BLUE}[*] بدء سكربت الاستطلاع المتقدم وفحص الملفات الحساسة...${NC}"

# اطلب رابط الهدف من المستخدم عند التشغيل
read -p "أدخل رابط الموقع المستهدف (مثال: https://target.com): " TARGET

# إزالة الشرطة المائلة الزائدة في نهاية الرابط إن وجدت
TARGET="${TARGET%/}"

echo -e "\n${YELLOW}[1] فحص الملفات الحساسة ومحاولة قراءة الإصدارات الحقيقية...${NC}"

FILES=(
    "/core/changelog.txt"
    "/composer.json"
    "/robots.txt"
    "/web.config"
    "/core/install.php"
    "/README.md"
)

for FILE in "${FILES[@]}"
do
    URL="${TARGET}${FILE}"
    # جلب كود الحالة الهيدر (HTTP Status)
    STATUS=$(curl -s -o /dev/null -w "%{http_code}" "$URL")
    
    if [ "$STATUS" -eq 200 ]; then
        echo -e "${GREEN}[+] تم العثور على الملف ($STATUS): $URL${NC}"
        # عرض أول 10 أسطر من الملف لمعاينة المحتوى سريعاً
        echo -e "${BLUE}--- معاينة المحتوى ---${NC}"
        curl -s "$URL" | head -n 10
        echo -e "${BLUE}----------------------${NC}\n"
    else
        echo -e "${RED}[-] غير موجود أو محمي ($STATUS): $FILE${NC}"
    fi
done

echo -e "\n${YELLOW}[2] فحص الـ CVEs الثنائية لـ Drupal 9.5.7 الموثقة أمنياً...${NC}"
echo -e "إصدار 9.5.7 يحتوي تاريخياً على نقاط ضعف يجب التحقق من ترقيعها يدوياً:"
echo -e "${RED}- CVE-2023-40619:${NC} ثغرة تخطي صلاحيات (Access Bypass) في نواة دروبال 9.5.x."
echo -e "${RED}- CVE-2023-27350:${NC} ثغرة Open Redirect قد تستغل في هجمات الهندسة الاجتماعية."

echo -e "\n${GREEN}[+] الخطوة المقترحة الآن:${NC}"
echo -e "إذا وجد السكربت ملف ${BLUE}/core/changelog.txt${NC} مفتوحاً، قم بقراءته بالكامل لتأكيد إذا كان السيرفر يعمل بـ 9.5.7 فعلياً أم إصدار آخر."
