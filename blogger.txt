from curl_cffi import requests
import time
import statistics
import random

class AbyssDecisionEngine:
    def __init__(self, target):
        self.target = target
        self.stats = {"baseline": {}, "scans": []}
        self.confidence_score = 0.0

    def get_fingerprint_behavior(self, path):
        """
        تحليل متقدم للاستجابات (Header, Size, Time, Structure)
        يستهدف التفريق بين الـ WAF والـ Backend.
        """
        profiles = ["chrome120", "firefox120", "safari15_5"]
        results = []

        print(f"[*] Decision Engine: Analyzing behavioral variance for {path}...")
        
        for profile in profiles:
            start = time.perf_counter()
            try:
                # محاكاة HTTP/2 و TLS الحديثة
                resp = requests.get(
                    f"https://{self.target}{path}", 
                    impersonate=profile, 
                    verify=False, 
                    timeout=12
                )
                duration = time.perf_counter() - start
                
                analysis = {
                    "profile": profile,
                    "status": resp.status_code,
                    "length": len(resp.content),
                    "headers_count": len(resp.headers),
                    "server": resp.headers.get("Server", "Unknown"),
                    "latency": duration,
                    "body_hash": hash(resp.text[:500]) # مقارنة بنية الرد
                }
                results.append(analysis)
            except Exception as e:
                print(f"[-] Profile {profile} failed: {e}")

        return self.infer_source(results)

    def infer_source(self, results):
        """
        محرك الاستنتاج الأمني (Inference Engine)
        يميز بين WAF Block و Rate Limit و Backend
        """
        if not results: return "No Data"

        statuses = [r['status'] for r in results]
        lengths = [r['length'] for r in results]
        latencies = [r['latency'] for r in results]

        # 1. تحليل التباين (Variance Analysis)
        # إذا كان حجم الرد ثابت تماماً رغم اختلاف الـ Status Code، فهذا WAF Static Page
        size_variance = statistics.variance(lengths) if len(lengths) > 1 else 0
        
        # 2. الاستنتاج المنطقي
        if all(s == 406 for s in statuses):
            self.confidence_score = 0.9
            return "RESULT: Confirmed WAF Block (Strict JA3/JA4 Fingerprinting Policy)"
        
        if any(s == 429 for s in statuses):
            return "RESULT: Rate Limiting Detected (IP/Behavioral Reputation)"

        if size_variance > 100:
            self.confidence_score = 0.7
            return "RESULT: Potential Backend Access (Content varies by Fingerprint)"

        return "RESULT: Indeterminate (Possibly Global Interception)"
