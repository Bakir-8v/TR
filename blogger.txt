import asyncio
import re
from curl_cffi.requests import AsyncSession

class AbyssFinalEngine:
    def __init__(self, target_ip):
        self.target_url = f"https://{target_ip}"
        # اختيار بصمات حديثة جداً لتجاوز JA3/JA4
        self.profiles = ["chrome124", "safari17_0", "chrome110"]

    async def probe(self, path):
        url = f"{self.target_url}{path}"
        print(f"[*] Analyzing: {path}")
        
        # استخدام Session للحفاظ على الكوكيز وإدارة HTTP/2
        async with AsyncSession(verify=False) as session:
            for profile in self.profiles:
                print(f"[*] Impersonating: {profile}")
                try:
                    # محاكاة كاملة للمتصفح (TLS + Headers + HTTP/2)
                    response = await session.get(url, impersonate=profile, timeout=20)
                    
                    status = response.status_code
                    print(f"[+] Status: {status} | Size: {len(response.content)}")

                    if status == 200:
                        print(f"[!!!] BREACH SUCCESSFUL WITH {profile}!")
                        title = re.search('<title>(.*?)</title>', response.text, re.I)
                        print(f"[*] Title: {title.group(1) if title else 'N/A'}")
                        return # توقف عند النجاح
                    
                    elif status == 406:
                        print(f"[-] 406 Reject: The WAF signature for {profile} is still detected.")
                    
                    elif status == 403:
                        print(f"[-] 403 Forbidden: WAF bypassed, but access denied by ACL.")

                except Exception as e:
                    print(f"[-] Error testing {profile}: {str(e)}")

async def main():
    TARGET = "94.55.118.66"
    engine = AbyssFinalEngine(TARGET)
    
    # اختبار المسارات التي أعطت 406 سابقاً
    for path in ["/admin", "/v1/api"]:
        await engine.probe(path)
        print("-" * 30)

if __name__ == "__main__":
    # تشغيل المحرك بطريقة متوافقة مع asyncio الحديث
    try:
        asyncio.run(main())
    except KeyboardInterrupt:
        pass
    except Exception as e:
        print(f"[!] System Error: {e}")
