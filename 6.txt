    <div class="header-box">
        <h2 style="margin:0;">نظام إدارة ملفات TR 📂</h2>

    <div id="status" class="status-bar">جاهز...</div>

    <textarea id="editor" placeholder="جاري تحميل المحتوى من المستودع..."></textarea>

        <button class="save-btn" onclick="saveData()">حفظ التعديلات (Write) 💾</button>
        <button class="load-btn" onclick="loadData()">تحديث البيانات (Read) 🔄</button>

<script>
    const CONFIG = {
        TOKEN: "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",
        OWNER: "Bakir-8v",
        REPO: "TR",
        PATH: "blogger.txt"
    };

    let currentSHA = "";

    // الدالة الأولى: القراءة (Read)
    async function loadData() {
        const status = document.getElementById('status');
        const editor = document.getElementById('editor');
        const shaDisplay = document.getElementById('sha-info');

        status.innerHTML = '<span class="loading">⏳ جاري القراءة من المستودع...</span>';

        try {
            const url = `https://api.github.com/repos/${CONFIG.OWNER}/${CONFIG.REPO}/contents/${CONFIG.PATH}`;
            const response = await fetch(url, {
                headers: { "Authorization": `token ${CONFIG.TOKEN}` }
            });

            if (!response.ok) throw new Error("الملف غير موجود أو الرابط خاطئ");

            const data = await response.json();
            currentSHA = data.sha;
            shaDisplay.innerText = `SHA: ${currentSHA}`;

            // فك التشفير من Base64 مع دعم اليونيكود
            const decodedContent = decodeURIComponent(escape(atob(data.content)));
            editor.value = decodedContent;

            status.innerHTML = '<span class="success">✅ تمت القراءة بنجاح!</span>';
        } catch (err) {
            status.innerHTML = `<span class="error">❌ خطأ في القراءة: ${err.message}</span>`;
        }
    }

    // الدالة الثانية: الكتابة (Write)
    async function saveData() {
        const status = document.getElementById('status');
        const content = document.getElementById('editor').value;

        status.innerHTML = '<span class="loading">⏳ جاري الكتابة والحفظ...</span>';

        try {
            const url = `https://api.github.com/repos/${CONFIG.OWNER}/${CONFIG.REPO}/contents/${CONFIG.PATH}`;
            
            // تحويل النص إلى Base64 مع دعم اليونيكود
            const encodedContent = btoa(unescape(encodeURIComponent(content)));

            const response = await fetch(url, {
                method: "PUT",
                headers: {
                    "Authorization": `token ${CONFIG.TOKEN}`,
                    "Content-Type": "application/json"
                },
                body: JSON.stringify({
                    message: "تعديل بواسطة محرر الويب",
                    content: encodedContent,
                    sha: currentSHA // مهم جداً لتحديث الملف
                })
            });

            if (!response.ok) throw new Error("فشل الحفظ، ربما بسبب تزامن الـ SHA");

            const result = await response.json();
            currentSHA = result.content.sha; // تحديث الـ SHA الجديد بعد الحفظ
            document.getElementById('sha-info').innerText = `SHA: ${currentSHA}`;

            status.innerHTML = '<span class="success">✅ تم الحفظ والتحديث في المستودع بنجاح!</span>';
        } catch (err) {
            status.innerHTML = `<span class="error">❌ خطأ في الحفظ: ${err.message}</span>`;
        }
    }

    // تشغيل القراءة تلقائياً عند فتح الصفحة
    window.onload = loadData;
</script>
